project(ztsilib)
add_library(ztsilib STATIC Ztsi.cpp)
target_link_libraries(ztsilib PRIVATE logging commonutils)
target_include_directories(ztsilib
    PUBLIC
        ${MODULES_INC_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
