project(pmc)

add_library(pmc SHARED PmcModule.cpp)
target_link_libraries(pmc
    PRIVATE
        commonutils
        logging
    PUBLIC
        pmclib
)

    PUBLIC
        ${MODULES_INC_DIR}
)

    PROPERTIES
        PREFIX ""
        POSITION_INDEPENDENT_CODE ON
)
